CREATE TABLE T_ARTISTS(
ID_ARTIST INTEGER PRIMARY KEY,
NAME TEXT,
ID_SPOTIFY_ARTIST TEXT
);


CREATE TABLE T_ALBUMS (
  ID_ALBUM INTEGER PRIMARY KEY,
  NAME TEXT,
  RELEASE_DATE DATE,
  TOTAL_TRACKS INTEGER,
  ID_SPOTIFY_ALBUM TEXT,
  ID_ARTIST INTEGER,
  CONSTRAINT FK_ARTISTS
  FOREIGN KEY (ID_ARTIST)
  REFERENCES T_ARTISTS(ID_ARTIST)
);

CREATE TABLE T_MUSICS (
  ID_MUSIC INTEGER PRIMARY KEY,
  NAME_MUSIC TEXT,
  EXPLICIT BOOL,
  DURATION_MS INTEGER,
  TRACK_NUMBER INTEGER,
  ID_SPOTIFY_MUSIC TEXT,
  ID_ALBUM INTEGER,
  CONSTRAINT FK_ALBUM
  FOREIGN KEY (ID_ALBUM)
  REFERENCES T_ALBUMS(ID_ALBUM)
);

CREATE TABLE T_EXECUTED(
PLAYED_AT DATETIME,
ID_MUSIC INTEGER,
CONSTRAINT FK_MUSIC
FOREIGN KEY (ID_MUSIC)
REFERENCES T_MUSICS(ID_MUSIC)
);


CREATE TABLE T_MUSIC_HAS_ARTISTS (
ID_MUSIC INTEGER,
ID_ARTIST INTEGER,
CONSTRAINT FK_ARTIST
FOREIGN KEY (ID_ARTIST)
REFERENCES T_ARTIST(ID_ARTIST),
CONSTRAINT FK_MUSIC
FOREIGN KEY (ID_MUSIC)
REFERENCES T_MUSICS(ID_MUSIC)
);
